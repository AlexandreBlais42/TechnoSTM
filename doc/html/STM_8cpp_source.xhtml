101bool STM::goToNextPosition() {
102 Vector3D<uint16_t> position = Plateforme::position;
103 bool direction = position.y & 0x01 ? false : true;
104 // Faux est vert la gauche et vrai vert la droite.
105 // On veut scanner en faisant un serpentin
106 if (position.x == resolutionX - 1 && position.y == resolutionY - 1) {
107 // Coin en bas à droite atteint, fin du scan
108 return false;
109 }
110 if (position.x == resolutionX - 1 && direction == true) {
111 position.y += 1;
112 } else if (position.x == 0 && direction == false) {
113 position.y += 1;
114 } else {
115 position.x += direction ? 1 : -1;
116 }
117 Plateforme::setPositionAbsolute(position);
118 return true;
119}
